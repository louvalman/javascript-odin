      let capitalize = (textcap) => {
        let firstletter = textcap.slice(0, 1).toUpperCase();
        let restofstring = textcap.slice(1).toLowerCase();
        return `parameter has been transformed into: ${
          firstletter + restofstring
        }. This was done by using the slice method to find the first letter of the string, and the remainder of the string, and using the toUpperCase and toLowerCase methods`;
      };
      console.log(capitalize("tHaNks fOr cHeCkIng mY aSsIgNmeNts oUt"));
      
  
      let lastLetter = (textlast) => {
        if (typeof textlast !== "string") {
          return `Error: Input must be a string`;
        }
        let finalassign = String(textlast.slice(-1));
        return `lastLetter is returning the last letter of the parameter: ${finalassign} - function always returns last letter of a string, or an error message`;
      };
